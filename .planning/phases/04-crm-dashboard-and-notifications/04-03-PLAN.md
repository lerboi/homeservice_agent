**Create `src/app/api/cron/send-recovery-sms/route.js`:**

```javascript
import { supabase } from '@/lib/supabase';
import { sendCallerRecoverySMS } from '@/lib/notifications';

export async function GET(request) {
  // Verify cron secret (Vercel sends CRON_SECRET header)
  const authHeader = request.headers.get('authorization');
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return Response.json({ error: 'Unauthorized' }, { status: 401 });
  }

  // Find calls that:
  // 1. Ended more than 60 seconds ago
  // 2. Have no linked appointment (caller hung up without booking)
  // 3. Haven't received recovery SMS yet
  // 4. Have a tenant and caller phone number
  // 5. Have status 'analyzed' (fully processed)
  const cutoff = new Date(Date.now() - 60_000).toISOString();

  const { data: calls, error } = await supabase
    .from('calls')
    .select('id, retell_call_id, from_number, to_number, tenant_id, start_timestamp, end_timestamp, retell_metadata')
    .eq('status', 'analyzed')
    .is('recovery_sms_sent_at', null)
    .lt('end_timestamp', cutoff)
    .not('tenant_id', 'is', null)
    .not('from_number', 'is', null)
    .limit(10); // Process max 10 per minute to avoid rate limits

  if (error || !calls?.length) {
    return Response.json({ processed: 0 });
  }

  let sent = 0;
  for (const call of calls) {
    // Calculate duration, skip short calls
    const duration = call.start_timestamp && call.end_timestamp
      ? Math.round((new Date(call.end_timestamp) - new Date(call.start_timestamp)) / 1000)
      : 0;
    if (duration < 15) {
      // Mark as processed so we don't re-check
      await supabase.from('calls').update({ recovery_sms_sent_at: new Date().toISOString() }).eq('id', call.id);
      continue;
    }

    // Check if appointment exists for this call (booked = no recovery needed)
    const { data: appt } = await supabase
      .from('appointments')
      .select('id')
      .eq('retell_call_id', call.retell_call_id)
      .maybeSingle();

    if (appt) {
      // Booked call — no recovery needed, mark as processed
      await supabase.from('calls').update({ recovery_sms_sent_at: new Date().toISOString() }).eq('id', call.id);
      continue;
    }

    // Look up tenant for business name and owner phone
    const { data: tenant } = await supabase
      .from('tenants')
      .select('business_name, owner_phone, retell_phone_number')
      .eq('id', call.tenant_id)
      .single();

    if (!tenant) continue;

    try {
      const callerName = call.retell_metadata?.caller_name || call.retell_metadata?.call_analysis?.caller_name || null;
      const appUrl = process.env.NEXT_PUBLIC_APP_URL || 'https://localhost:3000';

      await sendCallerRecoverySMS({
        to: call.from_number,
        callerName,
        businessName: tenant.business_name || 'us',
        bookingLink: `${appUrl}/book/${call.tenant_id}`,
        ownerPhone: tenant.owner_phone || tenant.retell_phone_number,
      });

      await supabase.from('calls').update({ recovery_sms_sent_at: new Date().toISOString() }).eq('id', call.id);
      sent++;
    } catch (err) {
      console.error(`Recovery SMS failed for call ${call.id}:`, err);
    }
  }

  return Response.json({ processed: calls.length, sent });
}
```
