**Create `src/lib/notifications.js`** with lazy-instantiated clients (same pattern as layer2-llm.js getClient()):

```javascript
import twilio from 'twilio';
import { Resend } from 'resend';
import { NewLeadEmail } from '@/emails/NewLeadEmail';

let twilioClient = null;
function getTwilioClient() {
  if (!twilioClient) {
    twilioClient = twilio(process.env.TWILIO_ACCOUNT_SID, process.env.TWILIO_AUTH_TOKEN);
  }
  return twilioClient;
}

let resendClient = null;
function getResendClient() {
  if (!resendClient) {
    resendClient = new Resend(process.env.RESEND_API_KEY);
  }
  return resendClient;
}
```

**Export sendOwnerSMS({ to, businessName, callerName, jobType, urgency, address, callbackLink, dashboardLink }):**
- Constructs body: `${businessName}: New ${urgency} lead -- ${callerName || 'Unknown'}, ${jobType || 'General inquiry'} at ${address || 'No address'}. Call back: ${callbackLink} | Dashboard: ${dashboardLink}`
- Calls `getTwilioClient().messages.create({ body, from: process.env.TWILIO_FROM_NUMBER, to })`
- Wraps in try/catch, logs error but does not throw (notification failure should not crash webhook)

**Export sendOwnerEmail({ to, lead, businessName, dashboardUrl }):**
- Calls `getResendClient().emails.send({ from: process.env.RESEND_FROM_EMAIL || 'alerts@homeservice.ai', to, subject: 'New ${lead.urgency} lead -- ${lead.caller_name || "Unknown caller"}', react: NewLeadEmail({ lead, businessName, dashboardUrl }) })`
- Same try/catch non-throwing pattern

**Export sendCallerRecoverySMS({ to, callerName, businessName, bookingLink, ownerPhone }):**
- firstName = callerName?.split(' ')[0] || 'there'
- body: `Hi ${firstName}, thanks for calling ${businessName}. We'd love to help -- book online at ${bookingLink} or call us back anytime at ${ownerPhone}.`
- Calls `getTwilioClient().messages.create({ body, from: process.env.TWILIO_FROM_NUMBER, to })`
- Same try/catch pattern

**Export sendOwnerNotifications({ tenantId, lead, businessName, ownerPhone, ownerEmail }):**
- Convenience wrapper that calls sendOwnerSMS + sendOwnerEmail in parallel via Promise.allSettled
- Constructs callbackLink as `tel:${lead.from_number}`
- Constructs dashboardLink as `${process.env.NEXT_PUBLIC_APP_URL || 'https://localhost:3000'}/dashboard/leads`
- Constructs dashboardUrl (same as dashboardLink for email)
- Logs results (which succeeded, which failed)
