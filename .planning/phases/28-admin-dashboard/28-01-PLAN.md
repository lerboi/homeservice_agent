```sql
CREATE TABLE admin_users (
```

```javascript
// Admin route gate — per D-03: middleware checks /admin/* by querying admin_users
const isAdminPath = pathname === '/admin' || pathname.startsWith('/admin/');

if (isAdminPath) {
  if (!user) {
    const signInUrl = new URL('/auth/signin', request.url);
    signInUrl.searchParams.set('redirect', pathname + request.nextUrl.search);
    return NextResponse.redirect(signInUrl);
  }
  // Check admin_users table — uses anon key client (user's own session)
  const { data: adminRecord } = await supabase
    .from('admin_users')
    .select('id')
    .eq('user_id', user.id)
    .single();

  if (!adminRecord) {
    // Rewrite to forbidden page — URL stays at attempted path (better UX)
    return NextResponse.rewrite(new URL('/admin/forbidden', request.url));
  }
  // Admin verified — continue to requested admin page
  return response;
}
```

IMPORTANT: The admin gate must `return response` after successful admin check to skip the onboarding/dashboard redirect logic below (admins may not have a tenant row).

2. **Create `src/lib/admin.js`** — shared helper for API routes to verify admin callers:

```javascript
import { createSupabaseServer } from '@/lib/supabase-server';
import { supabase } from '@/lib/supabase'; // service-role

/**
 * Verify the current request is from an admin user.
 * Uses the session client for auth, then service-role to check admin_users.
 * Returns the user object if admin, null otherwise.
 */
export async function verifyAdmin() {
  const supabaseUser = await createSupabaseServer();
  const { data: { user } } = await supabaseUser.auth.getUser();
  if (!user) return null;

  const { data } = await supabase
    .from('admin_users')
    .select('id, role')
    .eq('user_id', user.id)
    .single();

  return data ? user : null;
}
```
