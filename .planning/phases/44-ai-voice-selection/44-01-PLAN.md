    -- Migration 044: Add ai_voice column to tenants
    -- Phase 44: AI Voice Selection
    --
    -- Additive-only. ai_voice TEXT nullable — NULL means use VOICE_MAP[tone_preset] fallback.
    -- No backfill per D-17. Existing tenants keep NULL until they actively choose.

    **API route (src/app/api/ai-voice-settings/route.js):**

    Follow the exact pattern from `src/app/api/notification-settings/route.js`. Create a PATCH handler:

    ```javascript
    import { getTenantId } from '@/lib/get-tenant-id';
    import { supabase } from '@/lib/supabase';

    const VALID_VOICES = ['Aoede', 'Erinome', 'Sulafat', 'Zephyr', 'Achird', 'Charon'];

    export async function PATCH(request) {
      try {
        const tenantId = await getTenantId();
        if (!tenantId) {
          return Response.json({ error: 'Unauthorized' }, { status: 401 });
        }

        const { ai_voice } = await request.json();

        if (!VALID_VOICES.includes(ai_voice)) {
          return Response.json({ error: 'Invalid voice selection' }, { status: 400 });
        }

        const { error } = await supabase
          .from('tenants')
          .update({ ai_voice })
          .eq('id', tenantId);

        if (error) {
          return Response.json({ error: error.message }, { status: 500 });
        }

        return Response.json({ ai_voice });
      } catch (err) {
        return Response.json({ error: err.message }, { status: 500 });
      }
    }
    ```

    Export `VALID_VOICES` as a named export so the test can import it for parameterized validation.
