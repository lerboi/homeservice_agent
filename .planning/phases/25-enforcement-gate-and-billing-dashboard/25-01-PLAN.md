    **1. Create `/api/billing/invoices/route.js`** (new file):
    ```javascript
    import { stripe } from '@/lib/stripe';
    import { createSupabaseServer } from '@/lib/supabase-server';

    export async function GET() {
      const supabase = await createSupabaseServer();
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) return Response.json({ error: 'Unauthorized' }, { status: 401 });

      const { data: tenant } = await supabase
        .from('tenants').select('id').eq('owner_id', user.id).maybeSingle();
      if (!tenant) return Response.json({ error: 'No tenant' }, { status: 404 });

      const { data: sub } = await supabase
        .from('subscriptions')
        .select('stripe_customer_id')
        .eq('tenant_id', tenant.id)
        .eq('is_current', true)
        .maybeSingle();

      if (!sub?.stripe_customer_id) {
        return Response.json({ invoices: [] });
      }

      const invoices = await stripe.invoices.list({
        customer: sub.stripe_customer_id,
        limit: 5,
      });

      const formatted = invoices.data.map(inv => ({
        id: inv.id,
        date: inv.created,
        amount: inv.amount_paid || inv.total,
        currency: inv.currency,
        status: inv.status,
        hosted_invoice_url: inv.hosted_invoice_url,
      }));

      return Response.json({ invoices: formatted });
    }
    ```
