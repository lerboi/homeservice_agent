**Connect handler:**
```javascript
async function handleConnect(provider) {
  setConnecting(provider);
  const res = await fetch(`/api/accounting/${provider}/auth`);
  const { url, error } = await res.json();
  if (error) {
    toast.error(`Couldn't connect to ${provider}. Please try again.`);
    setConnecting(null);
    return;
  }
  window.location.href = url; // Redirect to OAuth consent
}
```

**Disconnect handler:**
```javascript
async function handleDisconnect(provider) {
  setDisconnecting(provider);
  await fetch('/api/accounting/disconnect', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ provider }),
  });
  // Remove from connections state
  setConnections(prev => { const next = {...prev}; delete next[provider]; return next; });
  setDisconnecting(null);
  setDisconnectTarget(null);
}
```
