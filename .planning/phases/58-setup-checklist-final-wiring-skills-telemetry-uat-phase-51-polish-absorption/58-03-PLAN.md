```python
"""Phase 58 CTX-01 — shared helpers for integration telemetry writes to activity_log.

Uses the REAL activity_log column names (event_type + metadata) per migration 004,
NOT the CONTEXT D-06 wording (action + meta). Documented in 58-RESEARCH §B.2.

Every helper is silent-on-failure: telemetry errors must never bubble up and
break the call path. Callers should fire these inside asyncio.gather (per-fetch)
or asyncio.create_task (fanout, fire-and-forget) so they don't serialize with
the primary return path.
"""
from __future__ import annotations

import asyncio
import logging
from typing import Any, Mapping, Optional

logger = logging.getLogger(__name__)


async def emit_integration_fetch(
    tenant_id: str,
    provider: str,
    duration_ms: int,
    cache_hit: bool,
    counts: Mapping[str, int],
    phone_e164: Optional[str],
) -> None:
    """Log one activity_log row for a single fetchCustomerByPhone call."""
    from ..supabase_client import get_supabase_admin

    def _insert() -> None:
        try:
            admin = get_supabase_admin()
            (
                admin.table("activity_log")
                .insert(
                    {
                        "tenant_id": tenant_id,
                        "event_type": "integration_fetch",
                        "metadata": {
                            "provider": provider,
                            "duration_ms": int(duration_ms),
                            "cache_hit": bool(cache_hit),
                            "counts": dict(counts),
                            "phone_e164": phone_e164,
                        },
                    }
                )
                .execute()
            )
        except Exception as exc:
            logger.warning("emit_integration_fetch failed: %s", exc)

    await asyncio.to_thread(_insert)


async def emit_integration_fetch_fanout(
    tenant_id: str,
    duration_ms: int,
    per_task_ms: Mapping[str, int],
    call_id: Optional[str],
) -> None:
    """Log one activity_log row for the pre-session asyncio.gather boundary."""
    from ..supabase_client import get_supabase_admin

    def _insert() -> None:
        try:
            admin = get_supabase_admin()
            (
                admin.table("activity_log")
                .insert(
                    {
                        "tenant_id": tenant_id,
                        "event_type": "integration_fetch_fanout",
                        "metadata": {
                            "duration_ms": int(duration_ms),
                            "per_task_ms": {k: int(v) for k, v in per_task_ms.items()},
                            "call_id": call_id,
                        },
                    }
                )
                .execute()
            )
        except Exception as exc:
            logger.warning("emit_integration_fetch_fanout failed: %s", exc)

    await asyncio.to_thread(_insert)
```

```python
import time
from .lib.telemetry import emit_integration_fetch_fanout

async def _timed(name: str, coro, per_task_ms: dict):
    _t0 = time.perf_counter()
    try:
        return await coro
    finally:
        per_task_ms[name] = int((time.perf_counter() - _t0) * 1000)

# Inside the function that holds the pre-session gather:
per_task_ms: dict[str, int] = {}
_fanout_start = time.perf_counter()
results = await asyncio.gather(
    _timed("sub", check_subscription(tenant_id), per_task_ms),
    _timed("intake", fetch_intake_config(tenant_id), per_task_ms),
    _timed("call", insert_call_row(tenant_id, caller, ...), per_task_ms),
    _timed("xero", fetch_xero_context_bounded(tenant_id, caller_e164), per_task_ms),
    _timed("jobber", fetch_jobber_context_bounded(tenant_id, caller_e164), per_task_ms),
    return_exceptions=True,
)
_fanout_duration_ms = int((time.perf_counter() - _fanout_start) * 1000)

# Fire-and-forget — do NOT await (telemetry must not delay session.start)
asyncio.create_task(
    emit_integration_fetch_fanout(
        tenant_id=tenant_id,
        duration_ms=_fanout_duration_ms,
        per_task_ms=per_task_ms,
        call_id=call_id,  # use whatever existing variable name holds the call UUID
    )
)
```

- Fanout telemetry is fire-and-forget via `asyncio.create_task` so it never delays `session.start`.
