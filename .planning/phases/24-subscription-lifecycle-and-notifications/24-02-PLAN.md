Actually, the simplest correct approach per the UI-SPEC: the right link is "Update Payment Method" pointing to a Stripe portal session. Since generating portal sessions requires server-side Stripe SDK, create a tiny API route:

Create `src/app/api/billing/portal/route.js`:
```js
import { stripe } from '@/lib/stripe';
import { createClient } from '@/lib/supabase-server';

export async function GET(request) {
  const supabase = createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return Response.json({ error: 'Unauthorized' }, { status: 401 });

  const { data: tenant } = await supabase
    .from('tenants').select('id').eq('owner_id', user.id).maybeSingle();
  if (!tenant) return Response.json({ error: 'No tenant' }, { status: 404 });

  const { data: sub } = await supabase
    .from('subscriptions').select('stripe_customer_id')
    .eq('tenant_id', tenant.id).eq('is_current', true).maybeSingle();
  if (!sub?.stripe_customer_id) return Response.json({ error: 'No subscription' }, { status: 404 });

  const session = await stripe.billingPortal.sessions.create({
    customer: sub.stripe_customer_id,
    return_url: `${process.env.NEXT_PUBLIC_APP_URL}/dashboard`,
  });

  return Response.redirect(session.url, 303);
}
```
