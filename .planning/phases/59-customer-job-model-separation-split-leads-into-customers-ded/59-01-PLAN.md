       ```js
       import { parsePhoneNumber, isValidPhoneNumber } from 'libphonenumber-js';
       export function normalizeE164(raw, countryHint = null) {
         if (!raw) throw new Error('phone_required');
         const parsed = parsePhoneNumber(raw, countryHint);
         if (!parsed || !parsed.isValid()) throw new Error('phone_invalid');
         return parsed.format('E.164');
       }
       export function isValidE164(s) { return typeof s === 'string' && /^\+[1-9]\d{6,14}$/.test(s); }
       export function formatInternational(e164) {
         const parsed = parsePhoneNumber(e164);
         return parsed ? parsed.formatInternational() : e164;
       }
       ```

    ```sql
    -- READ-ONLY AUDIT. Do not run via supabase db push. Run manually before 053a.
    -- Per D-13c (CONTEXT) this audit makes NO quality judgments — do not filter test/spam data.
    -- Backfill in Plan 02 processes every row as-is; owner deletes test customers/inquiries post-cutover.

    -- ===================================================================
    -- Pitfall 2 (A2): leads.from_number not in E.164
    -- ===================================================================
    SELECT COUNT(*) AS non_e164_phones FROM leads
    WHERE from_number IS NULL OR from_number !~ '^\+[1-9]\d{6,14}$';

    -- Sample the offenders
    SELECT DISTINCT from_number FROM leads
    WHERE from_number IS NULL OR from_number !~ '^\+[1-9]\d{6,14}$'
    LIMIT 50;

    -- ===================================================================
    -- Pitfall 1 (A3): invoices attached to leads with NULL appointment_id
    -- ===================================================================
    SELECT COUNT(*) AS invoices_for_unbooked_leads
    FROM invoices i
    JOIN leads l ON l.id = i.lead_id
    WHERE l.appointment_id IS NULL;

    -- Invoices with no lead at all (ad-hoc)
    SELECT COUNT(*) AS invoices_without_lead FROM invoices WHERE lead_id IS NULL;

    -- ===================================================================
    -- Expected customer count (distinct tenant/phone pairs) — D-13b dedup key
    -- ===================================================================
    SELECT COUNT(*) AS expected_customers FROM (
      SELECT DISTINCT tenant_id, from_number FROM leads WHERE from_number IS NOT NULL
    ) s;

    -- Leads with appointment_id (expected jobs — D-06 booked work only)
    SELECT COUNT(*) AS expected_jobs FROM leads WHERE appointment_id IS NOT NULL;

    -- Leads without appointment_id (expected inquiries — D-13a orphan branch)
    SELECT COUNT(*) AS expected_inquiries FROM leads WHERE appointment_id IS NULL;

    -- ===================================================================
    -- D-13a: Orphan leads (no appointment_id) broken down by their existing status.
    -- Each row becomes an Inquiry with status preserved verbatim (open → open, lost → lost, etc.).
    -- ===================================================================
    SELECT status, COUNT(*) AS orphan_leads_by_status
    FROM leads
    WHERE appointment_id IS NULL
    GROUP BY status
    ORDER BY orphan_leads_by_status DESC;

    -- ===================================================================
    -- D-13b: Duplicate-phone leads — groups where multiple leads share (tenant_id, phone).
    -- These collapse to ONE customer per group in Plan 02 backfill, with name/address from
    -- the MOST RECENT lead (ORDER BY created_at DESC LIMIT 1).
    -- ===================================================================
    SELECT COUNT(*) AS duplicate_phone_groups FROM (
      SELECT tenant_id, from_number, COUNT(*) AS lead_count
      FROM leads
      WHERE from_number IS NOT NULL
      GROUP BY tenant_id, from_number
      HAVING COUNT(*) > 1
    ) d;

    -- Total leads involved in duplicate-phone collapse (N leads → 1 customer each group)
    SELECT COALESCE(SUM(lead_count), 0) AS leads_in_duplicate_groups FROM (
      SELECT tenant_id, from_number, COUNT(*) AS lead_count
      FROM leads
      WHERE from_number IS NOT NULL
      GROUP BY tenant_id, from_number
      HAVING COUNT(*) > 1
    ) d;

    -- Sample of the largest duplicate-phone groups (verify they look reasonable — same person, not artifacts)
    SELECT tenant_id, from_number, COUNT(*) AS lead_count,
           array_agg(DISTINCT caller_name) AS names_seen,
           MIN(created_at) AS first_seen,
           MAX(created_at) AS last_seen
    FROM leads
    WHERE from_number IS NOT NULL
    GROUP BY tenant_id, from_number
    HAVING COUNT(*) > 1
    ORDER BY COUNT(*) DESC
    LIMIT 20;

    -- ===================================================================
    -- Cross-check: expected_customers + duplicate_phone_groups relationship
    -- expected_customers = (distinct tenant/phone pairs) — already collapsed.
    -- The difference (total leads - expected_customers) should equal
    -- (leads_in_duplicate_groups - duplicate_phone_groups) for internal consistency.
    -- ===================================================================
    SELECT
      (SELECT COUNT(*) FROM leads WHERE from_number IS NOT NULL) AS total_leads_with_phone,
      (SELECT COUNT(*) FROM (SELECT DISTINCT tenant_id, from_number FROM leads WHERE from_number IS NOT NULL) s) AS distinct_tenant_phone_pairs;
    ```
