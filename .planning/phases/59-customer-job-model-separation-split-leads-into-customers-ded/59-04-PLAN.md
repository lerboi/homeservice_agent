    export async function listCustomers({ tenantId, search }) {
      const supabase = await getServerSupabase();
      let q = supabase.from('customers')
        .select(`
          id, name, phone_e164, default_address, email, tags, created_at, updated_at,
          jobs:jobs(count),
          open_inquiries:inquiries(count)
        `)
        .eq('tenant_id', tenantId)
        .is('merged_into', null)
        .order('updated_at', { ascending: false });
      if (search) {
        const s = search.trim();
        q = q.or(`name.ilike.%${s}%,phone_e164.ilike.%${s}%`);
      }
      const { data, error } = await q;
      if (error) throw error;
      return data;
    }

    export async function getCustomerWithStats({ tenantId, customerId }) {
      const supabase = await getServerSupabase();
      const [{ data: customer, error: cErr }, { data: stats, error: sErr }] = await Promise.all([
        supabase.from('customers').select('*').eq('id', customerId).eq('tenant_id', tenantId).is('merged_into', null).single(),
        // Aggregate stats — live computed
        supabase.rpc('get_customer_stats', { p_customer_id: customerId }).maybeSingle().catch(() => ({ data: null, error: null })),

      if (cErr) throw cErr;
      // Fallback aggregate in-app if RPC not present (flagged as optional helper; compute inline if absent)
      const aggregate = stats ?? await computeStatsInline({ tenantId, customerId });
      return { customer, stats: aggregate };

    import { NextResponse } from 'next/server';
    import { getTenantId, getServerSupabase } from '@/lib/supabase-server';
    import { mergeCustomer } from '@/lib/customers';
    export async function POST(request, { params }) {
      const tenantId = await getTenantId();
      if (!tenantId) return NextResponse.json({ error: 'unauthorized' }, { status: 401 });
      // Resolve user id for D-19 audit merged_by
      const supabase = await getServerSupabase();
      const { data: userData } = await supabase.auth.getUser();
      const userId = userData?.user?.id ?? null;
      const { id } = await params;
      const body = await request.json();
      const target_id = body?.target_id;
      if (!target_id) return NextResponse.json({ error: 'target_id_required' }, { status: 400 });
      if (target_id === id) return NextResponse.json({ error: 'self_merge_forbidden' }, { status: 400 });
      try {
        const result = await mergeCustomer({ tenantId, sourceId: id, targetId: target_id, mergedBy: userId });
        return NextResponse.json(result);  // includes audit_id
      } catch (e) {
        const msg = String(e?.message ?? e);
        if (msg.includes('not_found_or_cross_tenant')) return NextResponse.json({ error: 'not_found' }, { status: 404 });
        if (msg.includes('source_invalid') || msg.includes('target_invalid')) return NextResponse.json({ error: msg }, { status: 404 });
        if (msg.includes('self_merge_forbidden')) return NextResponse.json({ error: msg }, { status: 400 });
        return NextResponse.json({ error: 'merge_failed', detail: msg }, { status: 500 });
      }
    }
