    ```javascript
    const { calculateAvailableSlots } = require('../../src/lib/scheduling/slot-calculator');

    describe('calculateAvailableSlots — time block exclusion', () => {
      const workingHours = {
        monday: { open: '09:00', close: '17:00' },
        tuesday: { open: '09:00', close: '17:00' },
        // ... all days
      };

      const baseParams = {
        workingHours,
        slotDurationMins: 60,
        existingBookings: [],
        zones: [],
        zonePairBuffers: {},
        tenantTimezone: 'America/Chicago',
        maxSlots: 50,
      };

      test('time block from 12:00-13:00 excludes overlapping slot', () => {
        // Use a known Monday date
        const targetDate = '2026-04-13'; // Monday
        const externalBlocks = [{
          start_time: '2026-04-13T17:00:00Z', // 12:00 CT
          end_time: '2026-04-13T18:00:00Z',   // 13:00 CT
        }];

        const slots = calculateAvailableSlots({
          ...baseParams,
          externalBlocks,
          targetDate,
        });

        // No slot should overlap with 12:00-13:00 CT
        for (const slot of slots) {
          const start = new Date(slot.start);
          const end = new Date(slot.end || slot.start);
          // The blocked window in UTC
          expect(start.getTime()).not.toBeLessThan(
            new Date('2026-04-13T18:00:00Z').getTime()
          );
        }
      });

      test('without time blocks, 12:00-13:00 slot is available', () => {
        const targetDate = '2026-04-13';
        const slots = calculateAvailableSlots({
          ...baseParams,
          externalBlocks: [],
          targetDate,
        });

        expect(slots.length).toBeGreaterThan(0);
      });

      test('all-day time block returns no available slots', () => {
        const targetDate = '2026-04-13';
        const externalBlocks = [{
          start_time: '2026-04-13T14:00:00Z', // 09:00 CT
          end_time: '2026-04-13T22:00:00Z',   // 17:00 CT
        }];

        const slots = calculateAvailableSlots({
          ...baseParams,
          externalBlocks,
          targetDate,
        });

        expect(slots).toHaveLength(0);
      });
    });
    ```

    2. Without blocks, slots in that window exist
