NOTE: Per D-01 (E.164 EXACT match, no normalization fallback) the implementation uses approach (1) from RESEARCH Pitfall 1: post-filter in JS. The Xero `where` filter uses `Phones[0].PhoneNumber.Contains("...")` with the last 10 digits as a HINT to narrow the candidate set — but final equality is enforced in JS. If the Contains hint returns 0, fall back to a broader unfiltered fetch with a stop-on-match (capped at 1 page = 100 contacts) to handle contacts whose phone is stored under index 1/2/3.
```javascript
/**
 * Fetch caller context from Xero by phone number (E.164).
 * Returns shape per Phase 55 D-02/D-03. Returns { contact: null } on no-match
 * or when the tenant has no Xero credentials (uniform with disconnected case).
 *
 * Cached via two-tier cacheTag (D-05):
 *   xero-context-${tenantId}                   (broad — invalidated on disconnect/reauth)
 *   xero-context-${tenantId}-${phoneE164}      (specific — invalidated by webhook)
 *
 * 'use cache' MUST be the FIRST statement (P54 known pitfall — silent disable otherwise).
 *
 * @param {string} tenantId   - Voco application tenant_id (UUID)
 * @param {string} phoneE164  - Caller phone in E.164 format (e.g. "+15551234567")
 * @returns {Promise<{contact: object|null, outstandingBalance?: number, lastInvoices?: object[], lastPaymentDate?: string|null}>}
 */
async fetchCustomerByPhone(tenantId, phoneE164) {
  'use cache';
  cacheTag(`xero-context-${tenantId}`);
  cacheTag(`xero-context-${tenantId}-${phoneE164}`);

  // Sanity-validate inputs (defense — also guards against cacheTag injection).
  if (typeof tenantId !== 'string' || typeof phoneE164 !== 'string') {
    return { contact: null };
  }
  if (!/^\+[1-9]\d{6,14}$/.test(phoneE164)) {
    return { contact: null };
  }

  const admin = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY,
  );

  const { data: cred } = await admin
    .from('accounting_credentials')
    .select('*')
    .eq('tenant_id', tenantId)
    .eq('provider', 'xero')
    .maybeSingle();
  if (!cred) return { contact: null };  // disconnected → empty context

  let refreshed;
  try {
    refreshed = await refreshTokenIfNeeded(admin, cred);
  } catch (err) {
    // Refresh failed — Plan 03/05 owns the error_state write + email surface.
    // From the call/dashboard read path we just degrade silently.
    return { contact: null };
  }
  this.setCredentials(refreshed);
  const xeroOrgId = refreshed.xero_tenant_id;
  if (!xeroOrgId) return { contact: null };

  // 1. Look up contact by phone (post-filter approach — Pitfall 1).
  // Xero's `Phones[N].PhoneNumber` filter requires indexed access and `Contains`
  // is the only reliably-supported operator. We pull a candidate set via Contains
  // on the last 10 digits, then enforce E.164 exact equality in JS across all
  // 4 phone slots (DEFAULT, MOBILE, FAX, DDI).
  const lastTen = phoneE164.replace(/\D/g, '').slice(-10);
  let contactsResp;
  try {
    contactsResp = await this._xeroClient.accountingApi.getContacts(
      xeroOrgId,
      undefined,                                            // ifModifiedSince
      `Phones[0].PhoneNumber.Contains("${lastTen}")`,        // narrow candidates
    );
  } catch (err) {
    return { contact: null };
  }
  let candidates = contactsResp.body?.contacts || [];
  const contact = candidates.find(c =>
    Array.isArray(c.phones) && c.phones.some(p => p.phoneNumber === phoneE164),
  );
  if (!contact) return { contact: null };

  // 2. Outstanding balance: AUTHORISED + AmountDue > 0
  let outstandingResp;
  try {
    outstandingResp = await this._xeroClient.accountingApi.getInvoices(
      xeroOrgId,
      undefined,
      `Status=="AUTHORISED" AND Contact.ContactID==guid("${contact.contactID}") AND AmountDue>0`,
    );
  } catch (err) {
    outstandingResp = { body: { invoices: [] } };
  }
  const outstandingInvoices = outstandingResp.body?.invoices || [];
  const outstandingBalance = outstandingInvoices.reduce(
    (sum, inv) => sum + (Number(inv.amountDue) || 0),
    0,
  );

  // 3. Last 3 invoices: AUTHORISED + PAID, Date DESC
  let recentResp;
  try {
    recentResp = await this._xeroClient.accountingApi.getInvoices(
      xeroOrgId,
      undefined,
      `(Status=="AUTHORISED" OR Status=="PAID") AND Contact.ContactID==guid("${contact.contactID}")`,
      'Date DESC',
      undefined,
      1,                                                    // page 1
    );
  } catch (err) {
    recentResp = { body: { invoices: [] } };
  }
  const allRecent = recentResp.body?.invoices || [];
  const lastInvoices = allRecent.slice(0, 3).map(inv => ({
    invoiceNumber: inv.invoiceNumber,
    date: inv.date,
    total: inv.total,
    amountDue: inv.amountDue,
    status: inv.status,
    reference: inv.reference,
  }));

  // 4. Last payment date: MAX(FullyPaidOnDate) across PAID
  const paidDates = allRecent
    .filter(inv => inv.status === 'PAID' && inv.fullyPaidOnDate)
    .map(inv => inv.fullyPaidOnDate);
  const lastPaymentDate = paidDates.length > 0
    ? paidDates.sort().at(-1)
    : null;

  // 5. Touch last_context_fetch_at (per-fetch cadence — Claude's discretion).
  // Cheap write; fire-and-forget OK because cache absorbs subsequent reads.
  await admin
    .from('accounting_credentials')
    .update({ last_context_fetch_at: new Date().toISOString() })
    .eq('id', cred.id);

  return { contact, outstandingBalance, lastInvoices, lastPaymentDate };
}
```
