```javascript
  // ============================================================
  // Phase 57: schedule mirror routing (JOBSCHED-01, D-16)
  // ============================================================
  if (JOBBER_VISIT_TOPICS.has(topic)) {
    try {
      if (topic === 'VISIT_DESTROY') {
        // Delete the mirror row; itemId IS the visit id.
        await admin
          .from('calendar_events')
          .delete()
          .eq('tenant_id', cred.tenant_id)
          .eq('provider', 'jobber')
          .eq('external_id', itemId);
      } else if (topic === 'VISIT_COMPLETE') {
        // Visit moved to completed — per D-06 NOT mirrored. Delete if present.
        await admin
          .from('calendar_events')
          .delete()
          .eq('tenant_id', cred.tenant_id)
          .eq('provider', 'jobber')
          .eq('external_id', itemId);
      } else if (topic === 'ASSIGNMENT_DESTROY') {
        // Parent visit may still exist with different assignees.
        // Re-fetch; applyJobberVisit will delete if bookable-filter excludes.
        const visit = await fetchJobberVisitById({ cred, id: itemId }).catch(() => null);
        if (visit) {
          const clientName = visit.job?.client?.name?.full ?? null;
          await applyJobberVisit({
            admin,
            tenantId: cred.tenant_id,
            visit,
            bookableUserIds: cred.jobber_bookable_user_ids ?? null,
            clientName,
          });
        }
      } else {
        // VISIT_CREATE / VISIT_UPDATE / ASSIGNMENT_CREATE / ASSIGNMENT_UPDATE
        // All re-fetch the parent visit by id and apply.
        const visit = await fetchJobberVisitById({ cred, id: itemId });
        if (visit) {
          const clientName = visit.job?.client?.name?.full ?? null;
          await applyJobberVisit({
            admin,
            tenantId: cred.tenant_id,
            visit,
            bookableUserIds: cred.jobber_bookable_user_ids ?? null,
            clientName,
          });
        }
      }
    } catch (err) {
      // Structured error log — never log cred/token material.
      console.error(JSON.stringify({
        scope: 'jobber-webhook-mirror',
        tenant_id: cred.tenant_id,
        topic,
        status: 'error',
      }));
      // Silent 200 — Jobber retries handled idempotently via upsert unique constraint.
    }
  }

  // ASSIGNMENT_CREATE on a visit we haven't seen → fetchJobberVisitById may be a Job
  // ID, not a Visit ID. If the above returns null visit, fall through. Acceptable —
  // the next VISIT_UPDATE (if any) will re-converge state; poll cron covers gap.
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Jobber cloud → Voco webhook endpoint | HMAC-SHA256 signed; raw body comparison; `JOBBER_CLIENT_SECRET` = OAuth client_secret |
| Webhook handler → Supabase | Service-role admin client, bypasses RLS (tenant_id from resolved cred row) |
| Webhook handler → Jobber GraphQL | Re-fetches visit by ID using resolved cred's access token |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-57-03-01 | Spoofing | Forged webhook payload | mitigate | HMAC verification with `crypto.timingSafeEqual` on raw body vs `JOBBER_CLIENT_SECRET` (preserved from P56). Test 6 asserts 401 on bad signature |
| T-57-03-02 | Replay | Duplicate webhook delivery | mitigate | `applyJobberVisit` uses upsert on UNIQUE(tenant_id, provider, external_id) — idempotent. Duplicate deliveries produce identical row state |
| T-57-03-03 | Information Disclosure | Cross-tenant calendar_events write | mitigate | Tenant resolved ONLY via `accounting_credentials.external_account_id` exact-match (P56 precedent); silent 200 on unknown account (Test 7). `applyJobberVisit` always passes the resolved tenant_id, never a payload-supplied value |
| T-57-03-04 | Tampering | Malicious visit id forces delete of unrelated row | mitigate | Delete scoped by (tenant_id, provider='jobber', external_id) — cross-tenant deletion impossible because tenant_id is server-resolved |
| T-57-03-05 | Information Disclosure | Access token in error logs | mitigate | Error logger emits ONLY `{scope, tenant_id, topic, status}`; no cred/token/response body in log line |
| T-57-03-06 | DoS | Webhook retry storm | accept | Silent 200 on all non-HMAC failures prevents Jobber retry amplification; existing P56 pattern |
</threat_model>
