    Animated floating language bubbles with gentle float animation (transform/opacity only per D-17). Shows bubbles for EN, ES, ZH, MS, and a "70+" bubble.

    ```jsx
    function LanguageHeroVisual() {
      return (
        <div className="relative w-48 h-32 md:w-56 md:h-36" aria-hidden="true">
          {[
            { label: 'EN', x: '10%', y: '15%', delay: '0s', size: 'size-10' },
            { label: 'ES', x: '55%', y: '5%', delay: '0.5s', size: 'size-10' },
            { label: 'ZH', x: '75%', y: '45%', delay: '1s', size: 'size-9' },
            { label: 'MS', x: '20%', y: '60%', delay: '1.5s', size: 'size-9' },
            { label: '70+', x: '50%', y: '55%', delay: '2s', size: 'size-12', accent: true },
          ].map((b) => (
            <div
              key={b.label}
              className={`absolute ${b.size} rounded-full flex items-center justify-center text-xs font-bold border-2 ${
                b.accent
                  ? 'bg-[#F97316] text-white border-[#F97316] shadow-[0_0_12px_rgba(249,115,22,0.3)]'
                  : 'bg-white text-[#475569] border-stone-200'
              }`}
              style={{
                left: b.x,
                top: b.y,
                animation: `langFloat 3s ease-in-out ${b.delay} infinite`,
              }}
            >
              {b.label}
            </div>
          ))}
          <style jsx>{`
            @media (prefers-reduced-motion: no-preference) {
              @keyframes langFloat {
                0%, 100% { transform: translateY(0); }
                50% { transform: translateY(-8px); }
              }
            }
          `}</style>
        </div>
      );
    }
    ```
