"""
Jobber integration — Phase 56 Plan 05.

Python counterpart to src/lib/integrations/jobber.js in the Voco monorepo.
Mirrors the return shape field-for-field so src/lib/customer_context.py (Plan 06)
can merge Jobber + Xero uniformly.

Key constraints (research Pitfalls 3, 7, 10):
- X-JOBBER-GRAPHQL-VERSION header REQUIRED on every request (400 otherwise)
- Refresh-token rotation is MANDATORY — every refresh returns a new refresh_token
  which MUST be persisted immediately (old refresh_token re-use fails silently)
- httpx.Timeout(connect=0.3, read=0.7) — socket-level self-terminate to avoid
  asyncio.wait_for leaking in-flight connections

Never raises, never logs token material. On any failure, returns None.
"""
from __future__ import annotations

import asyncio
import json
import logging
import time
from datetime import datetime, timezone
from typing import Any, Optional

import httpx

# Re-use existing helpers in the livekit-agent repo
from src.supabase_client import get_supabase_admin
from src.lib.phone import _normalize_phone  # or fallback to phonenumbers if Task 1 chose that

log = logging.getLogger(__name__)

JOBBER_GRAPHQL_URL = "https://api.getjobber.com/api/graphql"
JOBBER_TOKEN_URL = "https://api.getjobber.com/api/oauth/token"
JOBBER_API_VERSION = "2024-04-01"  # keep in sync with Next.js Plan 01

OUTSTANDING_STATUSES = {"AWAITING_PAYMENT", "BAD_DEBT", "PARTIAL", "PAST_DUE"}

# Keep query literal parity with Plan 01 (src/lib/integrations/jobber.js)
FETCH_QUERY = """
query FetchClientByPhone($phone: String!) {
  clients(first: 25, filter: { phoneNumber: $phone }) {
    nodes {
      id name
      emails { address }
      phones { number }
      jobs(first: 4, sort: [{ key: UPDATED_AT, direction: DESCENDING }]) {
        nodes {
          jobNumber title jobStatus startAt endAt
          visits(first: 1, filter: { status: UPCOMING }) { nodes { startAt } }
        }
      }
      invoices(first: 10) {
        nodes { invoiceNumber issuedDate amount amountOutstanding invoiceStatus }
      }
      visits(first: 1, sort: [{ key: COMPLETED_AT, direction: DESCENDING }], filter: { completed: true }) {
        nodes { endAt completedAt }
      }
    }
  }
}
"""


async def _refresh_and_persist_token(supabase, cred: dict, client_id: str, client_secret: str) -> Optional[dict]:
    """Refresh the Jobber token and write BOTH access_token AND refresh_token back.

    Jobber rotates refresh_token on every refresh (research Pitfall 3). Losing
    the new one = auth break on next cycle. Returns the fresh cred dict on
    success, None on failure.
    """
    try:
        async with httpx.AsyncClient(timeout=httpx.Timeout(connect=0.3, read=1.5)) as hx:
            resp = await hx.post(
                JOBBER_TOKEN_URL,
                data={
                    "grant_type": "refresh_token",
                    "client_id": client_id,
                    "client_secret": client_secret,
                    "refresh_token": cred["refresh_token"],
                },
                headers={"Accept": "application/json"},
            )
        if resp.status_code != 200:
            return None
        body = resp.json()
        new_access = body.get("access_token")
        new_refresh = body.get("refresh_token")
        if not new_access or not new_refresh:
            return None  # Missing rotation = contract violation; don't write
        # Decode JWT exp for expiry_date (ms)
        expiry_ms = _decode_jwt_exp_ms(new_access)
        expiry_iso = (
            datetime.fromtimestamp(expiry_ms / 1000, tz=timezone.utc).isoformat()
            if expiry_ms else None
        )
        # Atomic write-back
        supabase.from_("accounting_credentials").update({
            "access_token": new_access,
            "refresh_token": new_refresh,
            "expiry_date": expiry_iso,
        }).eq("tenant_id", cred["tenant_id"]).eq("provider", "jobber").execute()
        return {**cred, "access_token": new_access, "refresh_token": new_refresh, "expiry_date": expiry_iso}
    except Exception:
        # Non-fatal — caller returns None
        return None


def _decode_jwt_exp_ms(jwt: str) -> Optional[int]:
    try:
        import base64
        parts = jwt.split(".")
        if len(parts) < 2: return None
        # base64url decode; pad as needed
        pad = "=" * (-len(parts[1]) % 4)
        payload = json.loads(base64.urlsafe_b64decode(parts[1] + pad))
        return int(payload["exp"]) * 1000 if "exp" in payload else None
    except Exception:
        return None


def _jobber_match_phone(phones: list, target_e164: str) -> bool:
    for p in phones or []:
        normalized = _normalize_phone(p.get("number", ""), "US")
        if normalized and normalized == target_e164:
            return True
    return False


async def fetch_jobber_customer_by_phone(tenant_id: str, phone_e164: str) -> Optional[dict]:
    """Fetch Jobber customer context by phone.

    Mirrors src/lib/integrations/jobber.js `fetchJobberCustomerByPhone`.

    Args:
        tenant_id: Voco tenant_id (UUID)
        phone_e164: Caller phone in E.164 (e.g. "+15551234567")

    Returns:
        dict with keys {client, recentJobs, outstandingInvoices, outstandingBalance, lastVisitDate}
        or None on no-match / disconnected / error.
    """
    try:
        if not tenant_id or not phone_e164:
            return None

        supabase = get_supabase_admin()

        # Load Jobber credentials
        result = (
            supabase.from_("accounting_credentials")
            .select("*")
            .eq("tenant_id", tenant_id)
            .eq("provider", "jobber")
            .maybe_single()
            .execute()
        )
        cred = result.data if result else None
        if not cred:
            return None

        import os
        client_id = os.environ.get("JOBBER_CLIENT_ID", "")
        client_secret = os.environ.get("JOBBER_CLIENT_SECRET", "")

        access_token = cred.get("access_token")
        # Proactive refresh if expiry_date is in the past
        expiry_iso = cred.get("expiry_date")
        needs_refresh = False
        if expiry_iso:
            try:
                needs_refresh = datetime.fromisoformat(expiry_iso.replace("Z", "+00:00")) <= datetime.now(timezone.utc)
            except Exception:
                needs_refresh = False

        if needs_refresh:
            refreshed = await _refresh_and_persist_token(supabase, cred, client_id, client_secret)
            if not refreshed:
                return None
            access_token = refreshed["access_token"]
            cred = refreshed

        headers = {
            "Authorization": f"Bearer {access_token}",
            "X-JOBBER-GRAPHQL-VERSION": JOBBER_API_VERSION,
            "Content-Type": "application/json",
        }

        async def _do_request(hdrs):
            async with httpx.AsyncClient(timeout=httpx.Timeout(connect=0.3, read=0.7)) as hx:
                r = await hx.post(JOBBER_GRAPHQL_URL, json={"query": FETCH_QUERY, "variables": {"phone": phone_e164}}, headers=hdrs)
                return r

        resp = await _do_request(headers)

        if resp.status_code == 401:
            # Reactive refresh + single retry
            refreshed = await _refresh_and_persist_token(supabase, cred, client_id, client_secret)
            if not refreshed:
                return None
            headers["Authorization"] = f"Bearer {refreshed['access_token']}"
            resp = await _do_request(headers)

        if resp.status_code != 200:
            return None

        body = resp.json()
        if body.get("errors"):
            return None

        nodes = (body.get("data") or {}).get("clients", {}).get("nodes", [])
        matched = next((c for c in nodes if _jobber_match_phone(c.get("phones") or [], phone_e164)), None)
        if not matched:
            return None

        # Shape output to match Plan 01 Next.js field-for-field
        now = datetime.now(timezone.utc)
        jobs_raw = (matched.get("jobs") or {}).get("nodes") or []
        jobs = []
        for j in jobs_raw:
            next_visit = None
            vn = (j.get("visits") or {}).get("nodes") or []
            if vn:
                next_visit = vn[0].get("startAt")
            jobs.append({
                "jobNumber": j.get("jobNumber"),
                "title": j.get("title"),
                "status": j.get("jobStatus"),
                "startAt": j.get("startAt"),
                "endAt": j.get("endAt"),
                "nextVisitDate": next_visit,
            })
        # Sort: future nextVisitDate ASC first, then GraphQL-order (updatedAt DESC)
        def _future(j):
            nv = j.get("nextVisitDate")
            if not nv: return False
            try: return datetime.fromisoformat(nv.replace("Z", "+00:00")) >= now
            except Exception: return False
        jobs.sort(key=lambda j: (not _future(j), j.get("nextVisitDate") or ""))
        jobs = jobs[:4]

        invoice_nodes = (matched.get("invoices") or {}).get("nodes") or []
        outstanding = [inv for inv in invoice_nodes if inv.get("invoiceStatus") in OUTSTANDING_STATUSES]
        outstanding_balance = float(sum((inv.get("amountOutstanding") or 0) for inv in outstanding))
        outstanding_invoices = [{
            "invoiceNumber": inv.get("invoiceNumber"),
            "issuedAt": inv.get("issuedDate"),
            "amount": float(inv.get("amount") or 0),
            "amountOutstanding": float(inv.get("amountOutstanding") or 0),
            "status": inv.get("invoiceStatus"),
        } for inv in outstanding[:3]]

        vs = (matched.get("visits") or {}).get("nodes") or []
        last_visit_date = vs[0].get("endAt") if vs else None

        # Telemetry touch (best-effort; don't fail the fetch)
        try:
            supabase.from_("accounting_credentials").update({
                "last_context_fetch_at": datetime.now(timezone.utc).isoformat(),
            }).eq("id", cred["id"]).execute()
        except Exception:
            pass

        return {
            "client": {
                "id": matched.get("id"),
                "name": matched.get("name"),
                "email": (matched.get("emails") or [{}])[0].get("address"),
            },
            "recentJobs": jobs,
            "outstandingInvoices": outstanding_invoices,
            "outstandingBalance": outstanding_balance,
            "lastVisitDate": last_visit_date,
        }
    except Exception:
        # Never raise — the 800ms race in Plan 06 silently skips on None.
        return None
