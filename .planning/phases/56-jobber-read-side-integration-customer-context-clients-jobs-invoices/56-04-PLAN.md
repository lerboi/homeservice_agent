 * Send a Resend email to the tenant owner when Jobber token refresh fails.
 * Mirrors notifyXeroRefreshFailure (P55 D-14 carried forward).

export async function notifyJobberRefreshFailure(tenantId, ownerEmail) {
  if (!ownerEmail) return;
  try {
    const { default: JobberReconnectEmail } = await import('@/emails/JobberReconnectEmail');
    const resend = getResendClient();
    await resend.emails.send({
      from: process.env.RESEND_FROM_EMAIL || 'Voco <noreply@voco.live>',
      to: ownerEmail,
      subject: 'Your Jobber connection needs attention',
      react: JobberReconnectEmail({
        ctaUrl: (process.env.NEXT_PUBLIC_APP_URL || 'https://voco.live') + '/dashboard/more/integrations',
      }),
    });
  } catch (err) {
    // Non-fatal — log only the message (V7 — never echo err.response or cred fields)
    console.error('[notifyJobberRefreshFailure] send failed:', err?.message || 'unknown');
  }
}
