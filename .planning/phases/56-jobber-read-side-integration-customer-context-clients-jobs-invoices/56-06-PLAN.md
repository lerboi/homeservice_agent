def test_M2_only_jobber():
    jobber = {
        "client": {"id": "j1", "name": "John", "email": "j@e.com"},
        "recentJobs": [{"jobNumber": "J-1", "title": "AC install", "status": "upcoming", "startAt": None, "endAt": None, "nextVisitDate": "2026-04-20"}],
        "outstandingInvoices": [{"invoiceNumber": "INV-1", "issuedAt": "2026-04-01", "amount": 100, "amountOutstanding": 100, "status": "AWAITING_PAYMENT"}],
        "outstandingBalance": 100.0,
        "lastVisitDate": "2026-04-15",
    }
    r = merge_customer_context(jobber=jobber, xero=None)
    assert r["client"]["id"] == "j1"
    assert r["recentJobs"][0]["jobNumber"] == "J-1"
    assert r["lastVisitDate"] == "2026-04-15"
    assert r["outstandingBalance"] == 100.0  # fallback to Jobber
    assert r["_sources"]["client"] == "Jobber"
    assert r["_sources"]["outstandingBalance"] == "Jobber"
    assert "lastPaymentDate" not in r or r.get("lastPaymentDate") is None

"""Phase 56 Plan 06 — field-level merge for Jobber + Xero customer context.

Merge rule (CONTEXT D-07):
    Jobber wins: client, recentJobs, lastVisitDate
    Xero wins:   outstandingBalance, lastPaymentDate, lastInvoices

Source annotations (CONTEXT D-08) — provenance markers in _sources dict.
Returns None when both providers missed (CONTEXT D-11 — block omitted).
"""
from __future__ import annotations

from typing import Optional


def merge_customer_context(jobber: Optional[dict], xero: Optional[dict]) -> Optional[dict]:
    if not jobber and not xero:
        return None

    merged: dict = {}
    sources: dict = {}

    # --- client (Jobber wins; Xero.contact is the fallback with key rename) ---
    if jobber and jobber.get("client"):
        merged["client"] = jobber["client"]
        sources["client"] = "Jobber"
    elif xero and xero.get("contact"):
        # Normalize Xero's shape into Jobber's client shape
        contact = xero["contact"]
        merged["client"] = {
            "id": contact.get("contactID") or contact.get("id"),
            "name": contact.get("name") or " ".join(filter(None, [contact.get("firstName"), contact.get("lastName")])) or None,
            "email": contact.get("emailAddress"),
        }
        sources["client"] = "Xero"

    # --- recentJobs (Jobber-only; omitted entirely when Xero is the only match) ---
    if jobber and jobber.get("recentJobs"):
        merged["recentJobs"] = jobber["recentJobs"]
        sources["recentJobs"] = "Jobber"

    # --- lastVisitDate (Jobber-only) ---
    if jobber and jobber.get("lastVisitDate"):
        merged["lastVisitDate"] = jobber["lastVisitDate"]
        sources["lastVisitDate"] = "Jobber"

    # --- outstandingBalance (Xero wins; Jobber is fallback) ---
    x_bal = xero.get("outstandingBalance") if xero else None
    j_bal = jobber.get("outstandingBalance") if jobber else None
    if x_bal is not None:
        merged["outstandingBalance"] = x_bal
        sources["outstandingBalance"] = "Xero"
    elif j_bal is not None:
        merged["outstandingBalance"] = j_bal
        sources["outstandingBalance"] = "Jobber"

    # --- lastPaymentDate (Xero-only) ---
    if xero and xero.get("lastPaymentDate"):
        merged["lastPaymentDate"] = xero["lastPaymentDate"]
        sources["lastPaymentDate"] = "Xero"

    # --- lastInvoices (Xero wins; Jobber.outstandingInvoices is rename fallback) ---
    if xero and xero.get("lastInvoices"):
        merged["lastInvoices"] = xero["lastInvoices"]
        sources["lastInvoices"] = "Xero"
    elif jobber and jobber.get("outstandingInvoices"):
        merged["lastInvoices"] = jobber["outstandingInvoices"]
        sources["lastInvoices"] = "Jobber"

    if not merged:
        return None
    merged["_sources"] = sources
    return merged
