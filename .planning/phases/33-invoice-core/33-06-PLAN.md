**Create `src/app/api/invoices/[id]/pdf/route.js`:**

```javascript
import { renderToBuffer } from '@react-pdf/renderer';
import { InvoicePDF } from '@/lib/invoice-pdf';
import { createSupabaseServer } from '@/lib/supabase-server';
import { getTenantId } from '@/lib/get-tenant-id';

export async function GET(request, { params }) {
  const tenantId = await getTenantId();
  if (!tenantId) return Response.json({ error: 'Unauthorized' }, { status: 401 });

  const { id } = await params;
  const supabase = await createSupabaseServer();

  // Fetch invoice
  const { data: invoice } = await supabase
    .from('invoices').select('*').eq('id', id).eq('tenant_id', tenantId).single();
  if (!invoice) return Response.json({ error: 'Not found' }, { status: 404 });

  // Fetch line items
  const { data: lineItems } = await supabase
    .from('invoice_line_items').select('*').eq('invoice_id', id).order('sort_order', { ascending: true });

  // Fetch settings
  const { data: settings } = await supabase
    .from('invoice_settings').select('*').eq('tenant_id', tenantId).single();

  const buffer = await renderToBuffer(
    <InvoicePDF invoice={invoice} settings={settings || {}} lineItems={lineItems || []} />
  );

  return new Response(buffer, {
    headers: {
      'Content-Type': 'application/pdf',
      'Content-Disposition': `attachment; filename="invoice-${invoice.invoice_number}.pdf"`,
    },
  });
}
```
