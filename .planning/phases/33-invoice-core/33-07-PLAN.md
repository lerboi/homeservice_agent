import { Html, Head, Body, Container, Section, Text, Hr, Img } from '@react-email/components';

  return (

      <Head />
      <Body style={main}>

          {/* Business header — logo if available, business name */}

          </Text>

          {/* Key details table */}
          <Section style={detailsBox}>
            <Text style={detailRow}>Invoice #: {invoice.invoice_number}</Text>
            <Text style={detailRow}>Amount Due: ${Number(invoice.total).toFixed(2)}</Text>
            <Text style={detailRow}>Due Date: {invoice.due_date}</Text>
            {invoice.payment_terms && <Text style={detailRow}>Terms: {invoice.payment_terms}</Text>}
          </Section>

          <Hr style={hr} />

          {/* Footer — BUSINESS NAME ONLY, NO VOCO */}
          <Text style={footer}>
            Sent by {settings.business_name}
            {settings.phone && ` | ${settings.phone}`}
            {settings.email && ` | ${settings.email}`}
          </Text>
          {/* CRITICAL D-09: NO "Powered by Voco", NO getvoco.ai URLs, NO Voco branding */}

}

**Create `src/app/api/invoices/[id]/send/route.js`:**

POST handler that sends the invoice via email and optionally SMS.

```javascript
import { renderToBuffer } from '@react-pdf/renderer';
import { InvoicePDF } from '@/lib/invoice-pdf';
import InvoiceEmail from '@/emails/InvoiceEmail';
import { createSupabaseServer } from '@/lib/supabase-server';
import { getTenantId } from '@/lib/get-tenant-id';
```

Also import `getResendClient` and `getTwilioClient` from `@/lib/notifications`.

Steps:
1. Auth check (createSupabaseServer, getTenantId).
2. Fetch invoice, line_items, settings (same as PDF route).
3. Validate: invoice must have `customer_email`. If not, return 400: `{ error: 'Customer email required' }`.
4. Generate PDF buffer: `renderToBuffer(<InvoicePDF ... />)`.
5. Send email via Resend:
   ```javascript
   await getResendClient().emails.send({
     from: `${settings.business_name || 'Invoice'} <invoices@getvoco.ai>`,
     to: invoice.customer_email,
     subject: `Invoice ${invoice.invoice_number} from ${settings.business_name || 'Your Service Provider'}`,
     react: InvoiceEmail({ invoice, settings, lineItems }),
     attachments: [{ content: pdfBuffer, filename: `invoice-${invoice.invoice_number}.pdf` }],
   });
   ```
6. Parse request body for `{ send_sms: boolean }`. If `send_sms` is true and `invoice.customer_phone` exists:
   a. Fetch tenant's phone number: query `tenants` where `id = tenantId`, get `retell_phone_number`.
   b. Send SMS via Twilio (wrap in try/catch — SMS failure should NOT fail the send):
      ```javascript
      const smsBody = `${settings.business_name}: Invoice #${invoice.invoice_number} for $${Number(invoice.total).toFixed(2)} due ${invoice.due_date}. Full invoice sent to your email. Questions? Call ${settings.phone}`;
      await getTwilioClient().messages.create({
        body: smsBody,
        from: tenant.retell_phone_number,
        to: invoice.customer_phone,
      });
      ```
   c. If Twilio throws, log warning but don't fail the request.
7. Update invoice: set `status = 'sent'`, `sent_at = new Date().toISOString()`, `updated_at = now()`.
8. Return 200 with updated invoice.
