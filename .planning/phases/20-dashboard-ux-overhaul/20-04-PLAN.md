**Create `src/components/dashboard/DashboardTour.jsx`** (per D-07, D-08, D-09):

```jsx
'use client';

import { useState, useEffect, useCallback } from 'react';
import dynamic from 'next/dynamic';
import { useReducedMotion } from 'framer-motion';

// Dynamic import to avoid SSR issues with Joyride
const Joyride = dynamic(() => import('react-joyride'), { ssr: false });

const TOUR_OFFERED_KEY = 'gsd_tour_offered';

const STEPS = [
  {
    target: '[data-tour="home-page"]',
    content: 'This is your command center. Check setup progress, today\'s stats, and recent activity here.',
    disableBeacon: true,
    placement: 'bottom',
  },
  {
    target: '[data-tour="setup-checklist"]',
    content: 'Complete these items to get your AI receptionist ready for real calls.',
    placement: 'bottom',
  },
  {
    target: '[href="/dashboard/leads"]',
    content: 'View and manage all your leads and bookings in one place.',
    placement: 'right',
  },
  {
    target: '[href="/dashboard/calendar"]',
    content: 'See your upcoming appointments and manage your availability.',
    placement: 'right',
  },
  {
    target: '[href="/dashboard/services"]',
    content: 'Configure the services you offer and set escalation rules.',
    placement: 'right',
  },
  {
    target: '[href="/dashboard/settings"]',
    content: 'Adjust your AI greeting, working hours, and calendar connections.',
    placement: 'right',
  },
];

export default function DashboardTour() {
  const [run, setRun] = useState(false);
  const [showOffer, setShowOffer] = useState(false);
  const [isReady, setIsReady] = useState(false);
  const prefersReduced = useReducedMotion();

  // Check localStorage for first visit
  useEffect(() => {
    // Wait a tick for DOM to settle (Pitfall 2 — targets must exist)
    const timer = setTimeout(() => {
      setIsReady(true);
      if (!localStorage.getItem(TOUR_OFFERED_KEY)) {
        setShowOffer(true);
        localStorage.setItem(TOUR_OFFERED_KEY, '1');
      }
    }, 500);
    return () => clearTimeout(timer);
  }, []);

  // Wire up the Start Tour button in the top bar
  useEffect(() => {
    const btn = document.getElementById('start-tour-btn');
    if (btn) {
      // Make button visible (Plan 01 sets it hidden by default)
      btn.classList.remove('hidden');
      btn.classList.add('flex');

      const handleClick = () => {
        setShowOffer(false);
        setRun(true);
      };
      btn.addEventListener('click', handleClick);
      return () => btn.removeEventListener('click', handleClick);
    }
  }, []);

  const handleCallback = useCallback((data) => {
    const { status } = data;
    if (status === 'finished' || status === 'skipped') {
      setRun(false);
      setShowOffer(false);
    }
  }, []);

  const startTour = () => {
    setShowOffer(false);
    setRun(true);
  };

  const dismissOffer = () => {
    setShowOffer(false);
  };

  return (
    <>
      {/* Welcome offer modal — shown once on first visit (per D-08) */}
      {showOffer && isReady && (
        <div className="fixed inset-0 z-[9998] flex items-center justify-center bg-black/30">
          <div className="bg-white rounded-2xl shadow-xl p-6 max-w-sm mx-4 text-center">
            <h3 className="text-lg font-semibold text-[#0F172A] mb-2">Welcome to your dashboard!</h3>
            <p className="text-sm text-[#475569] mb-5">Want a quick tour of the essentials? It takes less than a minute.</p>
            <div className="flex gap-3 justify-center">
              <button
                onClick={dismissOffer}
                className="px-4 py-2 text-sm font-medium text-[#475569] hover:text-[#0F172A] transition-colors"
              >
                Skip
              </button>
              <button
                onClick={startTour}
                className="px-4 py-2 text-sm font-medium text-white bg-[#C2410C] hover:bg-[#C2410C]/90 rounded-lg transition-colors"
              >
                Start Tour
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Joyride tour (per D-07, D-09) */}
      {isReady && (
        <Joyride
          steps={STEPS}
          run={run}
          continuous
          showSkipButton
          showProgress
          disableScrolling={false}
          disableAnimation={prefersReduced}
          styles={{
            options: {
              primaryColor: '#C2410C',
              backgroundColor: '#FFFFFF',
              textColor: '#0F172A',
              zIndex: 9999,
            },
            buttonNext: {
              backgroundColor: '#C2410C',
            },
            buttonBack: {
              color: '#475569',
            },
          }}
          callback={handleCallback}
        />
      )}
    </>
  );
}
```

Key details:
- Dynamic import for Joyride with `ssr: false` to avoid SSR window errors.
- `isReady` state gates Joyride rendering until DOM is settled (Pitfall 2 from RESEARCH.md).
- `TOUR_OFFERED_KEY = 'gsd_tour_offered'` — consistent with project's `gsd_` prefix pattern.
- Welcome offer modal: centered white card with "Start Tour" and "Skip" buttons (per D-08).
- Tour steps target `data-tour` attributes and `href` selectors (per Pitfall 4 — avoids sidebar-only targets).
- `disableAnimation={prefersReduced}` propagates reduced motion to Joyride (Pitfall 5).
- Tour button in top bar wired via DOM query on mount (the button is in layout.js, tour state is in this component).
- Brand orange `#C2410C` for spotlight and buttons (per D-09).
- `showSkipButton` included (per D-09).
- Max 2 sentences per step content (per D-09).
