**Create `src/components/dashboard/ProgressRing.jsx`** (per D-06):

A `'use client'` component that renders a CSS conic-gradient ring with two segments.

Props: `requiredComplete` (number 0-4), `requiredTotal` (4), `recommendedComplete` (number 0-2), `recommendedTotal` (2).

```jsx
'use client';

export default function ProgressRing({ requiredComplete, requiredTotal, recommendedComplete, recommendedTotal }) {
  const total = requiredTotal + recommendedTotal; // 6
  const requiredPct = (requiredComplete / total) * 100;
  const recommendedPct = (recommendedComplete / total) * 100;
  const totalPct = requiredPct + recommendedPct;
  const completedCount = requiredComplete + recommendedComplete;

  const style = {
    background: `conic-gradient(
      #C2410C 0% ${requiredPct}%,
      #78716C ${requiredPct}% ${requiredPct + recommendedPct}%,
      #E7E5E4 ${totalPct}% 100%
    )`,
  };

  return (
    <div className="relative h-16 w-16 rounded-full shrink-0" style={style}>
      <div className="absolute inset-[6px] rounded-full bg-white flex items-center justify-center">
        <span className="text-sm font-semibold text-[#0F172A]">{completedCount}/{total}</span>
      </div>
    </div>
  );
}
```

**Rewrite `src/components/dashboard/ChecklistItem.jsx`** (per D-04, D-05):

Props: `item` (from API), `type` ('required' | 'recommended'), `description` (string), `cta` ({ label, href } | undefined).

The component is expandable when the item is NOT complete. Complete items show collapsed with a checkmark.

```jsx
'use client';

import { useState } from 'react';
import { CheckCircle2, Circle, ChevronDown } from 'lucide-react';
import { AnimatePresence, motion, useReducedMotion } from 'framer-motion';
import Link from 'next/link';

export default function ChecklistItem({ item, type, description, cta }) {
  const [expanded, setExpanded] = useState(false);
  const prefersReduced = useReducedMotion();
  const isComplete = item.complete;

  // Badge styling per D-04
  const badgeClass = type === 'required'
    ? 'bg-[#C2410C]/10 text-[#C2410C] text-xs font-medium px-2 py-0.5 rounded-full'
    : 'bg-stone-100 text-stone-600 text-xs font-medium px-2 py-0.5 rounded-full';
  const badgeLabel = type === 'required' ? 'Required' : 'Recommended';

  return (
    <div className="border-b border-stone-100 last:border-b-0">
      {/* Row: icon + label + badge + chevron */}
      <button
        type="button"
        onClick={() => !isComplete && setExpanded(!expanded)}
        disabled={isComplete}
        className={`flex items-center gap-3 w-full py-3 min-h-[44px] text-left ${
          isComplete ? 'cursor-default' : 'cursor-pointer hover:bg-stone-50/50'
        } px-1 rounded-lg transition-colors`}
        aria-expanded={isComplete ? undefined : expanded}
      >
        {/* Check icon */}
        {isComplete ? (
          <CheckCircle2 className="h-5 w-5 text-[#C2410C] shrink-0" aria-hidden="true" />
        ) : (
          <Circle className="h-5 w-5 text-stone-300 shrink-0" aria-hidden="true" />
        )}

        {/* Label */}
        <span className={`flex-1 text-sm ${isComplete ? 'text-[#475569] line-through' : 'text-[#0F172A] font-medium'}`}>
          {item.label}
        </span>

        {/* Badge — shown only for incomplete items */}
        {!isComplete && (
          <span className={badgeClass}>{badgeLabel}</span>
        )}

        {/* Chevron — only for incomplete items */}
        {!isComplete && (
          <ChevronDown
            className={`h-4 w-4 text-stone-400 shrink-0 transition-transform duration-200 ${
              expanded ? 'rotate-180' : ''
            }`}
            aria-hidden="true"
          />
        )}
      </button>

      {/* Expandable description + CTA */}
      <AnimatePresence initial={false}>
        {expanded && !isComplete && (
          <motion.div
            initial={prefersReduced ? { opacity: 0 } : { height: 0, opacity: 0 }}
            animate={prefersReduced ? { opacity: 1 } : { height: 'auto', opacity: 1 }}
            exit={prefersReduced ? { opacity: 0 } : { height: 0, opacity: 0 }}
            transition={{ duration: 0.2, ease: 'easeOut' }}
            style={{ overflow: 'hidden' }}
          >
            <div className="pl-8 pr-2 pb-3">
              <p className="text-sm text-[#475569] mb-3">{description}</p>
              {cta && (
                <Link
                  href={cta.href}
                  className="inline-flex items-center px-3 py-1.5 text-sm font-medium text-white bg-[#C2410C] hover:bg-[#C2410C]/90 rounded-lg transition-colors"
                >
                  {cta.label}
                </Link>
              )}
            </div>
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
}
```

Key points:
- `AnimatePresence` with `initial={false}` prevents entry animation on mount (Pitfall 6 from RESEARCH).
- `height: 'auto'` animation via framer-motion handles variable content height correctly.
- `overflow: 'hidden'` set via `style` prop (not className) to work with framer-motion height animation.
- Complete items: `line-through` text, no badge, no chevron, not expandable.
- `min-h-[44px]` satisfies D-11 touch target.
- Badge colors: orange (#C2410C) for required, stone-100/stone-600 for recommended (per D-04).
