**Create `src/components/dashboard/QuickActionCards.jsx`** (per D-03):

A `'use client'` component that renders 2-3 contextual quick-action cards based on which checklist items are incomplete.

Props: `items` (array of checklist items from API response).

```jsx
'use client';

import Link from 'next/link';
import { Calendar, Clock, Phone, Wrench, UserCircle } from 'lucide-react';
import { motion, useReducedMotion } from 'framer-motion';
import { card } from '@/lib/design-tokens';

const QUICK_ACTIONS = {
  setup_profile: {
    title: 'Set Up Business Profile',
    description: 'Add your business name so the AI greets callers professionally.',
    href: '/dashboard/settings',
    icon: UserCircle,
    iconBg: 'bg-blue-50',
    iconColor: 'text-blue-600',
  },
  configure_services: {
    title: 'Add Your Services',
    description: 'List what you offer so the AI understands caller requests.',
    href: '/dashboard/services',
    icon: Wrench,
    iconBg: 'bg-amber-50',
    iconColor: 'text-amber-600',
  },
  make_test_call: {
    title: 'Make a Test Call',
    description: 'Hear your AI receptionist in action before going live.',
    href: '/dashboard/settings#ai',
    icon: Phone,
    iconBg: 'bg-green-50',
    iconColor: 'text-green-600',
  },
  connect_calendar: {
    title: 'Connect Your Calendar',
    description: 'Sync Google or Outlook so bookings appear in your calendar.',
    href: '/dashboard/settings#calendar',
    icon: Calendar,
    iconBg: 'bg-purple-50',
    iconColor: 'text-purple-600',
  },
  configure_hours: {
    title: 'Set Working Hours',
    description: 'Define when you are available so the AI books correctly.',
    href: '/dashboard/settings#hours',
    icon: Clock,
    iconBg: 'bg-rose-50',
    iconColor: 'text-rose-600',
  },
};

export default function QuickActionCards({ items }) {
  const prefersReduced = useReducedMotion();

  if (!items || items.length === 0) return null;

  // Get first 3 incomplete items that have quick actions defined
  const incompleteActions = items
    .filter(item => !item.complete && QUICK_ACTIONS[item.id])
    .slice(0, 3)
    .map(item => ({ ...QUICK_ACTIONS[item.id], id: item.id }));

  if (incompleteActions.length === 0) return null;

  return (
    <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4" data-tour="quick-actions">
      {incompleteActions.map((action, i) => {
        const Icon = action.icon;
        return (
          <motion.div
            key={action.id}
            initial={prefersReduced ? undefined : { opacity: 0, y: 8 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ duration: 0.25, delay: prefersReduced ? 0 : i * 0.05 }}
          >
            <Link
              href={action.href}
              className={`${card.base} ${card.hover} p-5 block group`}
            >
              <div className={`h-10 w-10 rounded-xl ${action.iconBg} flex items-center justify-center mb-3`}>
                <Icon className={`h-5 w-5 ${action.iconColor}`} />
              </div>
              <h3 className="text-sm font-semibold text-[#0F172A] mb-1">{action.title}</h3>
              <p className="text-xs text-[#475569] leading-relaxed">{action.description}</p>
            </Link>
          </motion.div>
        );
      })}
    </div>
  );
}
```

Key details:
- Cards use `card.base` + `card.hover` from design tokens (per D-15).
- Each action card has a colored icon background, title, and description.
- Only shows incomplete items, max 3.
- Returns null when all items are complete (cards disappear per D-03).
- Stagger animation on entrance, respects `prefers-reduced-motion`.
- `data-tour="quick-actions"` attribute for Joyride targeting in Plan 04.
